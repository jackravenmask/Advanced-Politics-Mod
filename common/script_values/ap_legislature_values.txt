law_approval = {
	if = {
		limit = { 
			law_stance = {
				law = scope:input_law.type
				value = strongly_disapprove
			}
		}
		value = -2
	}
	if = {
		limit = { 
			law_stance = {
				law = scope:input_law.type
				value = disapprove
			}
		}
		value = -1
	}
	if = {
		limit = { 
			law_stance = {
				law = scope:input_law.type
				value = neutral
			}
		}
		value = 0
	}
	if = {
		limit = { 
			law_stance = {
				law = scope:input_law.type
				value = approve
			}
		}
		value = 1
	}
	if = {
		limit = { 
			law_stance = {
				law = scope:input_law.type
				value = strongly_approve
			}
		}
		value = 2
	}	
}
